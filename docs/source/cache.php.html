class Suco_Cache
{
	/**
	 * 通过工厂单件实例化相关的缓存类
	 * 如:
	 * $c1 = Suco_Cache::factory('Memcache');
	 * $c1 = Suco_Cache::factory('File');
	 * 
	 * @return object
	 */
	public static function factory($adapter = 'file')
	{
		$class = 'Suco_Cache_'.ucfirst($adapter);
		return call_user_func (array($class, 'instance'));
	}
}
