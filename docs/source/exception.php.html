class Suco_Exception extends Exception
{
	/**
	 * 打印异常信息
	 * 部署模式下请勿使用此方法
	 *
	 * @return string
	 */
	public function dump()
	{
		return '
'
				. '
['.strtoupper(get_class($this)).']
'
				. ''.$this->getMessage().''
				. '
  • ' . str_replace("\n", '
  • ', $this->getTraceAsString()) . '
'
				. '
';
	}
}
